<body> <div id="get_palette"></div> //div선언 <script src="rk_colorPalette.js"></script> <script> //css호출 rk_colorPalette_css("rk_colorPalette_v1.0.css"); //원하는 위차에 palette 기능 불러오기 rk_colorPalette_init("get_palette"); </script> </body>
